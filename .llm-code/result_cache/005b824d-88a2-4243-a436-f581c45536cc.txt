1	"""記帳引擎"""
2	import uuid
3	from datetime import datetime, timedelta
4	from typing import Dict, List, Optional, Tuple
5	from .types import Transaction, TransactionType, Category, Budget, Account
6	from .storage import AccountingStorage
7	
8	
9	class AccountingEngine:
10	    """記帳引擎"""
11	    
12	    def __init__(self, storage: Optional[AccountingStorage] = None):
13	        self.storage = storage or AccountingStorage()
14	    
15	    def add_transaction(
16	        self,
17	        amount: float,
18	        transaction_type: TransactionType,
19	        category_id: str,
20	        account_id: str,
21	        description: str = "",
22	        date: Optional[datetime] = None,
23	        tags: Optional[List[str]] = None,
24	        notes: str = ""
25	    ) -> Transaction:
26	        """新增交易記錄"""
27	        transaction = Transaction(
28	            id=str(uuid.uuid4()),
29	            amount=amount,
30	            type=transaction_type,
31	            category_id=category_id,
32	            account_id=account_id,
33	            description=description,
34	            date=date or datetime.now(),
35	            tags=tags or [],
36	            notes=notes,
37	        )
38	        
39	        # 更新帳戶餘額
40	        self._update_account_balance(transaction)
41	        
42	        return self.storage.add_transaction(transaction)
43	    
44	    def _update_account_balance(self, transaction: Transaction):
45	        """更新帳戶餘額"""
46	        account = self.storage._accounts.get(transaction.account_id)
47	        if not account:
48	            return
49	        
50	        if transaction.type == TransactionType.INCOME:
51	            account.balance += transaction.amount
52	        else:
53	            account.balance -= transaction.amount
54	        
55	        self.storage._save_accounts()
56	    
57	    def get_summary(
58	        self,
59	        start_date: Optional[datetime] = None,
60	        end_date: Optional[datetime] = None
61	    ) -> Dict:
62	        """取得摘要統計"""
63	        transactions = self.storage.get_transactions(
64	            start_date=start_date,
65	            end_date=end_date
66	        )
67	        
68	        total_income = 0.0
69	        total_expense = 0.0
70	        category_summary: Dict[str, float] = {}
71	        
72	        for t in transactions:
73	            if t.type == TransactionType.INCOME:
74	                total_income += t.amount
75	            else:
76	                total_expense += t.amount
77	                category_summary[t.category_id] = category_summary.get(t.category_id, 0.0) + t.amount
78	        
79	        return {
80	            "total_income": total_income,
81	            "total_expense": total_expense,
82	            "net": total_income - total_expense,
83	            "category_breakdown": category_summary,
84	            "transaction_count": len(transactions),
85	        }
86	    
87	    def get_monthly_summary(self, year: int, month: int) -> Dict:
88	        """取得月度統計"""
89	        start_date = datetime(year, month, 1)
90	        if month == 12:
91	            end_date = datetime(year + 1, 1, 1) - timedelta(days=1)
92	        else:
93	            end_date = datetime(year, month + 1, 1) - timedelta(days=1)
94	        
95	        return self.get_summary(start_date, end_date)
96	    
97	    def get_category_expenses(
98	        self,
99	        start_date: Optional[datetime] = None,
100	        end_date: Optional[datetime] = None
101	    ) -> Dict[str, float]:
102	        """取得分類支出"""
103	        transactions = self.storage.get_transactions(
104	            start_date=start_date,
105	            end_date=end_date,
106	        )
107	        
108	        category_expenses: Dict[str, float] = {}
109	        for t in transactions:
110	            if t.type == TransactionType.EXPENSE:
111	                category_expenses[t.category_id] = category_expenses.get(t.category_id, 0.0) + t.amount
112	        
113	        return category_expenses
114	    
115	    def check_budget(
116	        self,
117	        category_id: str,
118	        start_date: Optional[datetime] = None,
119	        end_date: Optional[datetime] = None
120	    ) -> Tuple[float, float, bool]:
121	        """檢查預算使用情況
122	        
123	        回傳：(已使用金額，預算金額，是否超支)
124	        """
125	        budgets = self.storage.get_budgets(category_id)
126	        if not budgets:
127	            return (0.0, 0.0, False)
128	        
129	        # 找最適合的預算
130	        budget = None
131	        for b in budgets:
132	            if b.start_date <= (start_date or datetime.now()) <= (b.end_date or datetime.now()):
133	                budget = b
134	                break
135	        
136	        if not budget:
137	            return (0.0, 0.0, False)
138	        
139	        spent = sum(
140	            t.amount for t in self.storage.get_transactions(
141	                start_date=start_date,
142	                end_date=end_date,
143	                category_id=category_id,
144	            )
145	            if t.type == TransactionType.EXPENSE
146	        )
147	        
148	        return (spent, budget.amount, spent > budget.amount)
149	    
150	    def get_accounts_summary(self) -> Dict[str, float]:
151	        """取得帳戶總覽"""
152	        return {
153	            account.id: account.balance 
154	            for account in self.storage.get_accounts()
155	        }
156	    
157	    def get_total_balance(self) -> float:
158	        """取得總餘額"""
159	        return sum(
160	            account.balance 
161	            for account in self.storage.get_accounts()
162	        )
