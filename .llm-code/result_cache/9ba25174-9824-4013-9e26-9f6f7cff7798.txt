17	def make_plugin_dir(tmp_path: Path, data: dict, name: str = "my-plugin") -> Path:
18	    """Create a plugin directory with .claude-plugin/plugin.json."""
19	    plugin_dir = tmp_path / name
20	    manifest_dir = plugin_dir / ".claude-plugin"
21	    manifest_dir.mkdir(parents=True)
22	    (manifest_dir / "plugin.json").write_text(json.dumps(data))
23	    return plugin_dir

30	class TestPluginManifest:
31	    def test_from_path_basic(self, tmp_path: Path):
32	        data = {
33	            "name": "my-plugin",
34	            "version": "1.0.0",
35	            "description": "A test plugin",
36	        }
37	        plugin_dir = make_plugin_dir(tmp_path, data)
38	        manifest = PluginManifest.from_path(plugin_dir)
39	        assert manifest.name == "my-plugin"
40	        assert manifest.version == "1.0.0"
41	        assert manifest.description == "A test plugin"
42	
43	    def test_from_path_with_mcp_servers(self, tmp_path: Path):
44	        data = {
45	            "name": "mcp-plugin",
46	            "version": "2.0.0",
47	            "description": "Plugin with MCP",
48	            "mcpServers": [{"name": "server1", "command": "npx", "args": ["server1"]}],
49	        }
50	        plugin_dir = make_plugin_dir(tmp_path, data, name="mcp-plugin")

52	        assert manifest.name == "mcp-plugin"
53	        assert manifest.mcp_servers is not None

117	class TestInstalledPlugin:
118	    def test_installed_plugin_frozen(self, tmp_path: Path):
119	        import dataclasses
120	        data = {"name": "p", "version": "1.0.0", "description": "p"}
121	        plugin_dir = make_plugin_dir(tmp_path, data, name="p")

124	        with pytest.raises((dataclasses.FrozenInstanceError, AttributeError)):
125	            ip.enabled = False  # type: ignore[misc]

140	class TestPluginInstaller:
141	    def test_install_from_local(self, tmp_path: Path):
142	        install_dir = tmp_path / "installed"
143	        source_dir = tmp_path / "source"
144	        data = {"name": "test-plugin", "version": "1.0.0", "description": "Test"}
145	        make_plugin_dir(source_dir, data, name="")
146	        # source_dir itself is the plugin dir (has .claude-plugin/)
147	        (source_dir / ".claude-plugin" / "plugin.json").write_text(json.dumps(data))
148	
149	        installer = PluginInstaller(install_dir)
150	        result_path = installer.install_from_local(source_dir)
151	
152	        assert result_path.exists()
153	        assert (result_path / ".claude-plugin" / "plugin.json").exists()
154	
155	    def test_list_installed_with_state(self, tmp_path: Path):
156	        install_dir = tmp_path / "installed"
157	        data = {"name": "listed-plugin", "version": "1.0.0", "description": "Listed"}
158	        source_dir = tmp_path / "src"
159	        source_dir.mkdir()
160	        manifest_dir = source_dir / ".claude-plugin"
161	        manifest_dir.mkdir()
162	        (manifest_dir / "plugin.json").write_text(json.dumps(data))
163	
164	        installer = PluginInstaller(install_dir)
165	        installer.install_from_local(source_dir)
166	
167	        plugins = installer.list_installed()
168	        assert len(plugins) == 1
169	        assert plugins[0].manifest.name == "listed-plugin"
170	        assert plugins[0].enabled is True
171	
172	    def test_enable_disable(self, tmp_path: Path):
173	        install_dir = tmp_path / "installed"
174	        data = {"name": "toggle-plugin", "version": "1.0.0", "description": "Toggle"}
175	        source_dir = tmp_path / "src"
176	        source_dir.mkdir()
177	        manifest_dir = source_dir / ".claude-plugin"
178	        manifest_dir.mkdir()
179	        (manifest_dir / "plugin.json").write_text(json.dumps(data))
180	
181	        installer = PluginInstaller(install_dir)
182	        installer.install_from_local(source_dir)
183	
184	        installer.disable("toggle-plugin")
185	        plugins = installer.list_installed()
186	        assert plugins[0].enabled is False
187	
188	        installer.enable("toggle-plugin")
189	        plugins = installer.list_installed()
190	        assert plugins[0].enabled is True

192	    def test_uninstall_removes_dir(self, tmp_path: Path):
193	        install_dir = tmp_path / "installed"
194	        data = {"name": "removable", "version": "1.0.0", "description": "Remove me"}
195	        source_dir = tmp_path / "src"
196	        source_dir.mkdir()
197	        manifest_dir = source_dir / ".claude-plugin"
198	        manifest_dir.mkdir()
199	        (manifest_dir / "plugin.json").write_text(json.dumps(data))
200	
201	        installer = PluginInstaller(install_dir)
202	        installed_path = installer.install_from_local(source_dir)
203	
204	        assert installed_path.exists()
205	        installer.uninstall("removable")
